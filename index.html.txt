      <div class="block b3">
        שלם
        <input id="whole" type="number" placeholder="?" />
      </div>
    </div>

    <div id="feedback"></div>

    <div class="row">
      <button class="check" id="checkBtn">✓ בדיקה</button>
      <button class="again" id="againBtn" style="display:none">🔄 נסו שוב</button>

  <script>
    const questions = [
      { part1: 3, part2: 4, whole: 7 },
      { part1: 3, part2: 7, whole: 10 }
    ];

    let current = 0;

    const el = id => document.getElementById(id);
    const part1 = el('part1');
    const part2 = el('part2');
    const whole = el('whole');
    const feedback = el('feedback');
    const checkBtn = el('checkBtn');
    const againBtn = el('againBtn');

    function showMessage(type, text){
      feedback.innerHTML = `<div class="message ${type}">${text}</div>`;
    }

    function resetInputs(){
      part1.value = part2.value = whole.value = '';
      part1.disabled = part2.disabled = whole.disabled = false;
      againBtn.style.display = 'none';
      feedback.innerHTML = '';
    }

    function checkAnswer(){
      const q = questions[current];
      const p1 = parseInt(part1.value,10);
      const p2 = parseInt(part2.value,10);
      const w = parseInt(whole.value,10);

      if(isNaN(p1)||isNaN(p2)||isNaN(w)){
        showMessage('empty','נא למלא את כל המספרים ⚠️');
        return;
      }

      const correct = ((p1===q.part1 && p2===q.part2)||(p1===q.part2 && p2===q.part1)) && w===q.whole;

      if(correct){
        part1.disabled=part2.disabled=whole.disabled=true;
        showMessage('ok','כל הכבוד! כעת לחצו על חץ החזרה בג׳ניאלי וכתבו את הפתרון במקום המתאים כדי לעבור לחדר הבא.');
        return;
      }

      if(p1+p2===w){
        showMessage('warn','התרגיל המתמטי נכון, אבל לא מתאים לפתרון החידה. נסו שוב!');
        againBtn.style.display='inline-block';
        return;
      }

      showMessage('bad','אופס! נסו שוב 💪');
      againBtn.style.display='inline-block';
    }

    checkBtn.addEventListener('click',checkAnswer);
    againBtn.addEventListener('click',resetInputs);
  </script>
